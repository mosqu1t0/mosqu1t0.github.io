#include <bits/stdc++.h>

using namespace std;
const int N = 103;

int g[N][N];
int f[N][N];

int dx[4] = {-1, 0, 1, 0}, dy[4] = {0, 1, 0, -1};

int n, m;
int res;

int dp(int x, int y)
{
    if (f[x][y] != -1) return f[x][y];

    int ma = 0;
    for (int i = 0 ; i < 4 ; i ++)
    {
        int a = x + dx[i], b = y + dy[i];

        if (a < 1 || a > n || b < 1 || b > m) continue;

        if (g[x][y] > g[a][b]) ma = max(ma , dp(a, b));
    }

    return f[x][y] = ma + 1;
}


int main()
{
    cin >> n >> m;

    memset(f, -1, sizeof f);

    for (int i = 1; i <= n ; i ++)
        for (int j = 1; j <= m ; j ++)
            scanf("%d", &g[i][j]);

    for (int i = 1; i <= n ; i ++)
        for (int j = 1; j <= m ; j ++)
            res = max(dp(i, j), res);

    cout << res << endl;

    return 0;
}

include <bits/stdc++.h>
define LL __int128

using namespace std;
const int N = 83;
LL a[N][N];
LL f[N][N];
LL res;

int n, m;

inline LL
read()
{
    LL x = 0, f = 1;

    char ch = getchar();

    while (ch < '0' || ch > '9' )
    {
        if (ch == '-') f = -1;
        ch = getchar();
    }

    while (ch >= '0' && ch <= '9')
    {
        x = x * 10 + ch - '0';
        ch = getchar();
    }

    return x * f;
}

inline void
print(LL x)
{
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) print(x / 10);
    putchar(x % 10 + '0');
}

LL
dp(LL a[])
{
    memset(f, 0, sizeof f);

    for (int i = 0 ; i < m ; i ++)
        for (int j = 1 ; i + j <= m ; j ++)
            f[j][i+j] = max((f[j+1][i+j] + a[j]) << 1, (f[j][i+j-1]+a[i+j]) << 1);

    return f[1][m];
}

int
main()
{
    cin >> n >> m;

    for (int i = 1 ; i <= n ; i ++)
        for (int j = 1 ; j <= m ; j ++)
            a[i][j] = read();

    for (int i = 1 ; i <= n ; i ++) res += dp(a[i]);

    print(res);

    return 0;
}
