#include <iostream>

using namespace std;
const int N = 2e4 + 10, INF = 1e9;
int d[N][N];

int n, m, q;

void floyd(){
    for (int k = 1 ; k <= n ; k ++)
        for (int i = 1 ; i <= n ; i ++)
            for (int j = 1 ; j <= n ; j ++)
                if (d[i][j] > d[i][k] + d[k][j])
                    d[i][j] = d[i][k] + d[k][j];

}

int main()
{
    cin >> n >> m >> q;

    for (int i = 1 ; i <= n ; i ++)
        for (int j = 1 ; j <= n ; j ++)
            if (i == j) d[i][j] = 0;
            else d[i][j] = INF;

    while (m --){
        int a, b, c;
        cin >> a >> b >> c;
        d[a][b] = min(d[a][b], c);
    }

    floyd();

    while (q --){
        int a, b;
        cin >> a >> b;
        if (d[a][b] > INF / 2) puts("impossible");
        else cout << d[a][b] << endl;
    }

    return 0;
}

#include <iostream>
#include <cstring>

using namespace std;
const int N = 1e5 + 10;
int h[N], e[N], ne[N], idx;
int q[N], d[N], hh = 0, tt = -1;

int n, m;

void add(int a, int b)
{
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++;
}

bool topSort()
{
    for (int i = 1 ; i <= n ; i ++)
    {
        if (!d[i]) q[++ tt] = i;
    }

    while (hh <= tt)
    {
        int t = q[hh ++];

        for (int i = h[t] ; i != -1 ; i = ne[i])
        {
            int j = e[i];
            d[j] --;
            if (!d[j])
            {
                q[++ tt] = j;
            }
        }
    }
    if (tt == n - 1) return true;
    return false;
}

int main()
{
    cin >> n >> m;

    memset(h, -1, sizeof h);
    while (m --)
    {
        int a, b;
        cin >> a >> b;
        d[b] ++;
        add(a, b);
    }

    if (!topSort()) puts("-1");
    else
    {
        for (int i = 0 ; i < n; i ++)
        {
            if (i) cout << ' ';
            cout << q[i];
        }
    }

    return 0;
}
